function updateTimers(){ const elapsed = (Date.now()-startTime)/1000 events.forEach(e=>{ const remaining = e.time - elapsed const percent = Math.max(0,(remaining/e.time)*100) const bar = document.getElementById("bar-"+e.time) const label = document.getElementById("time-"+e.time) if(bar) bar.style.width = percent+"%" if(label){ if(remaining>0){ const min = Math.floor(remaining/60) const sec = Math.floor(remaining%60) label.innerText = `faltan ${min}:${sec.toString().padStart(2,"0")}` }else{ if(!label.dataset.done){ playBell(); label.dataset.done = "true"; } label.innerHTML = `LISTO` } } }) }
